1//! Cache maintenance operations
2//!
3//! These functions were adapted from the cortex-m (0.7.1) crate.
4//! cortex-m only lets you access these functions when you have
5//! the SCB in cortex_m::Peripherals collection. But, we neither want
6//! to steal the peripheral(s), nor own them. So, we're duplicating
7//! the routines that we need, and making sure that we're using
8//! them safely.
9//!
10//! cortex-m crate available at <https://github.com/rust-embedded/cortex-m>.
11//!
12//! See <https://github.com/rust-embedded/cortex-m/issues/304>, and also #239.
13//!
14//! <https://github.com/rust-embedded/cortex-m/pull/320> indicates that this might
15//! be available in a near-future cortex-m crate.
16
17/// Cleans and invalidates D-cache by address.
18///
19/// * `addr`: The address to clean and invalidate.
20/// * `size`: The number of bytes to clean and invalidate.
21///
22/// Cleans and invalidates D-cache starting from the first cache line containing `addr`,
23/// finishing once at least `size` bytes have been cleaned and invalidated.
24///
25/// It is recommended that `addr` is aligned to the cache line size and `size` is a multiple of
26/// the cache line size, otherwise surrounding data will also be cleaned.
27///
28/// Cleaning and invalidating causes data in the D-cache to be written back to main memory,
29/// and then marks that data in the D-cache as invalid, causing future reads to first fetch
30/// from main memory.
31pub fn clean_invalidate_dcache_by_address(addr: usize, size: usize) {
32    // No-op zero sized operations
33    if size == 0 {
34        return;
35    }
36
37    // Safety: write-only registers, pointer to static memory
38    let cbp = unsafe { &*cortex_m::peripheral::CBP::PTR };
39
40    cortex_m::asm::dsb();
41
42    // Cache lines are fixed to 32 bit on Cortex-M7 and not present in earlier Cortex-M
43    const LINESIZE: usize = 32;
44    let num_lines = ((size - 1) / LINESIZE) + 1;
45
46    let mut addr = addr & 0xFFFF_FFE0;
47
48    for _ in 0..num_lines {
49        // Safety: write to Cortex-M write-only register
50        unsafe { cbp.dccimvac.write(addr as u32) };
51        addr += LINESIZE;
52    }
53
54    cortex_m::asm::dsb();
55    cortex_m::asm::isb();
56}
